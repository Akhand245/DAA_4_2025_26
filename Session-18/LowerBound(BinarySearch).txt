int  lowerBound(int arr[], int size, int target){
    int l=0;
    int r=size-1;
    int ans=size;
    while(l<=r){
        int mid =l+(r-l)/2;
        if (arr[mid]>=target){
            ans = mid;
            r=mid-1;
        }else{
            l=mid+1;
        }
    }
    cout<<"lower bound index="<<ans<<endl;
    return ans;
}
